Exercise 3: Event-Driven Node.js

Terminal: node Exercise_3.js. Watch event cascade!

Exercise_3.js Code

const EventEmitter = require('events');
const myEmitter = new EventEmitter();

myEmitter.on('userLogin', (user) => console.log(`👤 ${user} logged in`));
myEmitter.on('orderPlaced', (orderId, amount) => console.log(`🛒 Order #${orderId}: $${amount}`));

myEmitter.emit('userLogin', 'alice@example.com');
myEmitter.emit('orderPlaced', 123, 99.99);
