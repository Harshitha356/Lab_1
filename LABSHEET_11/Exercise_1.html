Exercise 1: Node.js HTTP Server

Terminal: node Exercise_1.js, then test in browser http://localhost:3000/

Exercise_1.js Code

const http = require('http');

const server = http.createServer((req, res) => {
  res.setHeader('Content-Type', 'text/html');
  
  if (req.url === '/') {
    res.writeHead(200);
    res.write('
Node.js Server ✅
');
    res.end('
All requirements met!
');
  } else if (req.url === '/status') {
    res.writeHead(200);
    res.end('
Server Running!
');
  } else {
    res.writeHead(404);
    res.end('404 Not Found');
  }
});

server.listen(3000, () => {
  console.log('Server: http://localhost:3000/');
});
Click to test server (run node Exercise_1.js first!)
